#!/usr/bin/env perl

my %h=('a' => 1, 'b' => 2, 'c' => 3);

print "using foreach \n";
foreach (keys %h){
 print "$_ $h{$_} \n";
}

print "using while each \n";
while (my ($k,$v) = each %h){
 print "$k $v \n";
}
