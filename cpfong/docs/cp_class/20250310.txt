#!/usr/bin/env perl

my $foo = 'zxyabzdef';
my $pos = index($foo, 'd');

if ($pos > 0) {  # Ensure there is a character before 'd'
    my $char_before_d = substr($foo, $pos - 1, 1);
    print "Character before 'd' is: $char_before_d\n";  # Output: c
} else {
    print "char d is at the beginning or not found.\n";
}
